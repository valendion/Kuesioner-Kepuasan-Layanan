generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

/// The underlying table does not contain a valid unique identifier and can therefore currently not be handled by Prisma Client.
model patient {
  id                Int     @id @default(autoincrement())
  age               Int?
  gender            String? @db.VarChar(50)
  education         String? @db.VarChar(50)
  occupation        String? @db.VarChar(50)
  services_received String? @db.VarChar(50)
  answers           String? @db.VarChar(50)
  suggestion        String? @db.VarChar(512)
  createdAt         String? @db.VarChar(50)
  updatedAt         String? @db.VarChar(50)

  @@ignore
}
